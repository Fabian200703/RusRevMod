                target = RSA
                type = annex_everything
            }
			load_oob = SOV_1917
			195 = {
				create_unit = {
					division = "division_template = \"Opolcheniye Krasnoy Armii\""

					owner = SOV
				}
			}

country_event = {
	id = RSA_democratic_civil_war.002
	title = RSA_democratic_civil_war.002.t
	desc = RSA_democratic_civil_war.002.desc
	picture = GFX_report_event_generic_battle
	is_triggered_only = yes
	hidden = no




	#SOV bekommen am 07.11.17 195, 208, 244

	option = {
		name = RSA_democratic_civil_war.002.a
		257 = {
			transfer_state_to = GRA
		}
		GRA = {
			if = {
				limit = {
					RSA = {exists = yes}
				}
				declare_war_on = {
					target = RSA
					type = annex_everything
				}
			}

			if = {
				limit = {
					SOV = {exists = yes}
				}
				declare_war_on = {
					target = SOV
					type = annex_everything
				}
			}

			if = {
				limit = {
					BLA = {exists = yes}
				}
				declare_war_on = {
					target = BLA
					type = annex_everything
				}
			}
			257 = {
				create_unit = {
					division = "division_template = \"Kavaleriyskaya diviziya\""
					count = 10
					owner = SOV
				}
			}

			change_tag_from = RSA
			change_tag_from = SOV
			change_tag_from = BLA
		}
	}
}
